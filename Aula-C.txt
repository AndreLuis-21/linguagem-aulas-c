4) informe três notas e sua media caso aprovado

	#include <stdio.h>
	#include <stdlib.h>
	int main ()

	{
	
	float n1,n2,n3,m;
	
	printf("informe primeira nota:");
	scanf("%f",&n1);
	
	printf("informe segunda nota:");
	scanf("%f",&n2);
	
	printf("informe terceira nota:");
	scanf("%f",&n3);
	
	
	m=(n1+n2+n3)/3;
	
	if (m>=7)
	
	{
	printf("Aprovado");
	}
	
	else if(m>=5)
	{
	printf("Recuperação");
	}
	
	else
	{
	printf("Reprovado");
	}
	
	return 0;
